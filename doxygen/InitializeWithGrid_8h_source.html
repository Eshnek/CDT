108 template <typename OutputIt>
109 void generateGridTriangles(
110  OutputIt outFirst,
111  const IndexSizeType xres,
112  const IndexSizeType yres)
113 {
114  for(IndexSizeType iy = 0; iy < yres; ++iy)
115  {
116  for(IndexSizeType ix = 0; ix < xres; ++ix)
117  {
118  // 2___3 v3
119  // |\ | /\
120  // | \ | n3/ \n2
121  // |__\| /____\
122  // 0 1 v1 n1 v2
123  const IndexSizeType i = iy * xres + ix;
124  const IndexSizeType iv = iy * (xres + 1) + ix;
125  const VertInd vv[4] = {iv, iv + 1, iv + xres + 1, iv + xres + 2};
126  Triangle t;
127 
128  t.vertices = {vv[0], vv[1], vv[2]};
129  t.neighbors = {
130  iy ? 2 * i - xres * 2 + 1 : noNeighbor,
131  2 * i + 1,
132  ix ? 2 * i - 1 : noNeighbor};
133  *outFirst++ = t;
134 
135  t.vertices = {vv[1], vv[3], vv[2]};
136  t.neighbors = {
137  ix < xres - 1 ? 2 * i + 2 : noNeighbor,
138  iy < yres - 1 ? 2 * i + xres * 2 : noNeighbor,
139  2 * i};
140  *outFirst++ = t;
141  }
142  }
143 }
